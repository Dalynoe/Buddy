const fields=[

"candidate-name","badge-level","submission-date",

"project-summary","voluntary-summary",

"change-management","risk-management","stakeholder-management"

];

fields.forEach(id=>{

const el=document.getElementById(id);

const saved=localStorage.getItem(id);

if(saved) el.value=saved;

el.addEventListener("input",()=>{

localStorage.setItem(id,el.value);

showSave(id);

});

});

function showSave(id){

let section="candidate";

if(id.includes("project")) section="project";

else if(id.includes("voluntary")) section="voluntary";

else if(["change-management","risk-management","stakeholder-management"].includes(id)) section="areas";

const el=document.getElementById("save-status-"+section);

if(el){el.textContent="✅ Saved "+new Date().toLocaleTimeString();setTimeout(()=>el.textContent="",2500);}

}
